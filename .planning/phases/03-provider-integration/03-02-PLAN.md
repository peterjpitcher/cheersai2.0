**src/lib/providers/facebook/api.ts** -- Raw Graph API functions:
- `publishPagePost(pageId: string, accessToken: string, message: string): Promise<{ id: string }>` -- POST to `${getMetaGraphApiBase()}/${pageId}/feed` with `{ message, access_token }`
- `publishPagePhoto(pageId: string, accessToken: string, imageUrl: string, caption: string): Promise<{ id: string }>` -- POST to `${getMetaGraphApiBase()}/${pageId}/photos` with `{ url: imageUrl, caption, access_token }`
- `publishPageStory(pageId: string, accessToken: string, imageUrl: string): Promise<{ id: string }>` -- POST to `${getMetaGraphApiBase()}/${pageId}/photo_stories` with `{ photo_url: imageUrl, access_token }`
- All functions call `classifyMetaError` on non-OK responses and throw `ProviderError` with platform='facebook'
- Use fetch() with Content-Type application/json

**src/lib/providers/facebook/adapter.ts** -- FacebookAdapter class:

```typescript
export class FacebookAdapter implements PublishingAdapter {
  readonly platform = 'facebook' as const;
  
  supports(contentType: string): boolean {
    return ['instant_post', 'story', 'event', 'promotion'].includes(contentType);
  }
  
  validate(content: ContentPayload): ValidationResult {
    return validateFacebookContent(content);
  }
  
  async publishPost(connectionId: string, content: ContentPayload): Promise<PublishResult> {
    // 1. getDecryptedToken(connectionId, 'access') to get page token
    // 2. getConnectionMetadata(connectionId) to get pageId from metadata
    // 3. If content.mediaUrls?.length, call publishPagePhoto with first image
    // 4. Else call publishPagePost with text only
    // 5. Return { platformPostId: response.id }
  }
  
  async publishStory(connectionId: string, content: ContentPayload): Promise<PublishResult> {
    // 1. getDecryptedToken + getConnectionMetadata
    // 2. Require at least one media URL (validated earlier)
    // 3. Call publishPageStory with first image
    // 4. Return { platformPostId: response.id }
  }
}
```
