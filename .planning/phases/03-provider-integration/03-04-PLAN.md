**src/app/(app)/connections/actions.ts** -- Rewrite connection server actions for v2:

1. **`initiateOAuthConnect(provider: Provider)`** server action:
   - Call `requireAuthContext()` to get authenticated user
   - Generate cryptographically random state string: `crypto.randomUUID()`
   - Insert into `oauth_states` table: `{ state, provider, expires_at: new Date(Date.now() + 10 * 60 * 1000).toISOString(), created_by: user.id }`
   - Use service client for insert (since created_by uses auth.uid() but we need to ensure it works)
   - Call `buildOAuthRedirectUrl(provider, state)` to get redirect URL
   - Return `{ success: true, redirectUrl }`

2. **`completeOAuthConnect(provider: Provider, code: string, state: string)`** server action:
   - Call `requireAuthContext()` to get `accountId`
   - Validate state: query `oauth_states` where `state = state AND provider = provider AND used_at IS NULL AND expires_at > now()`
   - If no matching state found: return `{ success: false, error: 'Invalid or expired OAuth state' }`
   - Mark state as used: update `oauth_states` SET `used_at = now()` WHERE matching row
   - Call `exchangeProviderAuthCode(provider, code)` to exchange auth code for tokens
   - Upsert `social_connections`: `{ account_id: accountId, platform: provider, platform_account_id: result.metadata?.pageId ?? result.metadata?.igBusinessId ?? result.metadata?.locationId ?? 'default', platform_account_name: result.displayName, status: 'active', scopes: [...], token_expires_at: result.expiresAt, metadata: result.metadata, display_name: result.displayName, last_synced_at: new Date().toISOString() }`
   - Use UPSERT on `(account_id, platform, platform_account_id)` unique constraint
   - After upsert, get the connection ID from the upserted row
   - Store tokens in vault: `storeEncryptedToken(connectionId, 'access', result.accessToken)`
   - If result.refreshToken: `storeEncryptedToken(connectionId, 'refresh', result.refreshToken)`
   - Call `revalidatePath('/connections')` and `revalidatePath('/')` (for sidebar health dots)
   - Return `{ success: true }`

3. **`disconnectProvider(provider: Provider)`** server action:
   - Call `requireAuthContext()` to get `accountId`
   - Update `social_connections` SET `status = 'revoked'` WHERE `account_id = accountId AND platform = provider`
   - Token vault entries auto-cascade on connection delete (but we're updating status, not deleting)
   - Call `revalidatePath('/connections')`
   - Return `{ success: true }`

**src/lib/connections/health.ts** -- New file for health derivation (per D-01, PLAT-06):

```typescript
import type { ConnectionHealth, ConnectionHealthSummary, ProviderPlatform } from '@/types/providers';
import { requireAuthContext } from '@/lib/auth/server';

const EXPIRY_WARNING_DAYS = 7;
const EXPIRY_WARNING_MS = EXPIRY_WARNING_DAYS * 24 * 60 * 60 * 1000;

// Facebook page tokens don't expire (Research pitfall 3)
const NEVER_EXPIRING_PLATFORMS: ProviderPlatform[] = ['facebook'];

export function deriveConnectionHealth(
  status: string,
  tokenExpiresAt: string | null,
  platform: ProviderPlatform,
): ConnectionHealth {
  // Red: revoked or expired status
  if (status === 'revoked' || status === 'expired') return 'red';
  
  // If no expiry date:
  // - Facebook page tokens don't expire -> green
  // - Others -> amber (unknown expiry is a warning)
  if (!tokenExpiresAt) {
    return NEVER_EXPIRING_PLATFORMS.includes(platform) ? 'green' : 'amber';
  }
  
  const expiresAtMs = new Date(tokenExpiresAt).getTime();
  const now = Date.now();
  
  // Red: token already expired
  if (expiresAtMs <= now) return 'red';
  
  // Amber: within 7-day warning window
  if ((expiresAtMs - now) <= EXPIRY_WARNING_MS) return 'amber';
  
  // Green: active and not near expiry
  return 'green';
}

export async function getConnectionHealthSummaries(): Promise<ConnectionHealthSummary[]> {
  const { supabase, accountId } = await requireAuthContext();
  const { data, error } = await supabase
    .from('social_connections')
    .select('id, platform, platform_account_name, status, token_expires_at, last_synced_at')
    .eq('account_id', accountId)
    .order('platform');
  
  if (error) throw error;
  if (!data?.length) return [];
  
  return data.map(row => ({
    provider: row.platform as ProviderPlatform,
    health: deriveConnectionHealth(row.status, row.token_expires_at, row.platform as ProviderPlatform),
    accountName: row.platform_account_name ?? null,
    lastSyncedAt: row.last_synced_at ?? null,
    tokenExpiresAt: row.token_expires_at ?? null,
    connectionId: row.id,
  }));
}
```
