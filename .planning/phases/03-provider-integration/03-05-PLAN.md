**src/lib/providers/rate-limits.ts** -- Database-backed rate limit counters (per PLAT-08, Research pitfall 6):

```typescript
import type { ProviderPlatform } from '@/types/providers';
import { createServiceSupabaseClient } from '@/lib/supabase/service';

// Platform-specific rate limits
const RATE_LIMITS: Record<ProviderPlatform, { ceiling: number; windowMs: number }> = {
  facebook: { ceiling: 200, windowMs: 60 * 60 * 1000 },      // 200/hr (BUC policy)
  instagram: { ceiling: 200, windowMs: 60 * 60 * 1000 },     // 200/hr
  gbp: { ceiling: 1000, windowMs: 24 * 60 * 60 * 1000 },     // ~1000/day (varies by quota)
};

function getWindowStart(platform: ProviderPlatform): string {
  const windowMs = RATE_LIMITS[platform].windowMs;
  const now = Date.now();
  const windowStart = new Date(now - (now % windowMs));
  return windowStart.toISOString();
}

export async function incrementRateLimit(
  accountId: string,
  platform: ProviderPlatform,
  endpoint: string,
): Promise<void> {
  const supabase = createServiceSupabaseClient();
  const windowStart = getWindowStart(platform);
  const ceiling = RATE_LIMITS[platform].ceiling;
  
  // Upsert: increment if exists, create if not
  const { error } = await supabase.rpc('increment_rate_limit', {
    p_account_id: accountId,
    p_provider: platform,
    p_endpoint: endpoint,
    p_window_start: windowStart,
    p_limit_ceiling: ceiling,
  });
  
  // Fallback if RPC not available: manual upsert
  if (error) {
    await supabase.from('provider_rate_limits').upsert(
      {
        account_id: accountId,
        provider: platform,
        endpoint,
        window_start: windowStart,
        request_count: 1,
        limit_ceiling: ceiling,
      },
      { onConflict: 'account_id,provider,endpoint,window_start' },
    );
  }
}

export async function checkRateLimit(
  accountId: string,
  platform: ProviderPlatform,
  endpoint: string,
): Promise<{ allowed: boolean; remaining: number; retryAfterMs?: number }> {
  const supabase = createServiceSupabaseClient();
  const windowStart = getWindowStart(platform);
  const ceiling = RATE_LIMITS[platform].ceiling;
  
  const { data } = await supabase
    .from('provider_rate_limits')
    .select('request_count')
    .eq('account_id', accountId)
    .eq('provider', platform)
    .eq('endpoint', endpoint)
    .eq('window_start', windowStart)
    .single();
  
  const currentCount = data?.request_count ?? 0;
  const remaining = Math.max(0, ceiling - currentCount);
  
  if (remaining <= 0) {
    const windowMs = RATE_LIMITS[platform].windowMs;
    const windowStartMs = new Date(windowStart).getTime();
    const retryAfterMs = (windowStartMs + windowMs) - Date.now();
    return { allowed: false, remaining: 0, retryAfterMs: Math.max(0, retryAfterMs) };
  }
  
  return { allowed: true, remaining };
}

export async function getRateLimitStatus(
  accountId: string,
  platform: ProviderPlatform,
): Promise<{ endpoint: string; count: number; ceiling: number }[]> {
  const supabase = createServiceSupabaseClient();
  const windowStart = getWindowStart(platform);
  
  const { data } = await supabase
    .from('provider_rate_limits')
    .select('endpoint, request_count, limit_ceiling')
    .eq('account_id', accountId)
    .eq('provider', platform)
    .eq('window_start', windowStart);
  
  return (data ?? []).map(row => ({
    endpoint: row.endpoint,
    count: row.request_count,
    ceiling: row.limit_ceiling,
  }));
}
```

**src/app/api/cron/token-health/route.ts** -- Nightly token health cron (per PLAT-10):

This endpoint is called by QStash on a nightly schedule. It:

1. **Validates CRON_SECRET** from Authorization header (timing-safe comparison using existing pattern from notify-expiring-connections route)
2. **Queries all social_connections** across all accounts via service client:
   ```sql
   SELECT id, account_id, platform, status, token_expires_at, platform_account_name
   FROM social_connections
   WHERE status != 'revoked'
   ```
3. **For each connection**, calls `deriveConnectionHealth(status, token_expires_at, platform)`:
   - If health is 'red' and status is not already 'expired': update status to 'expired'
   - If health is 'amber': log warning with provider and account details
   - If health is 'green': no action (skip)
4. **Logs summary**: `{ checked: N, healthy: N, warning: N, expired: N }`
5. **Returns** 200 with JSON summary

Follow the exact CRON_SECRET validation pattern from `src/app/api/cron/notify-expiring-connections/route.ts`:
- Read `CRON_SECRET` from env
- Compare against `Authorization: Bearer {secret}` header
- Return 401 if mismatch

Export only `GET` handler (cron endpoints use GET).

The cron schedule configuration (QStash) is handled externally -- this plan only creates the endpoint.

**src/features/connections/health-dots.tsx** -- Sidebar health dot indicators (per D-01):

Server component that renders three small coloured circles next to the Connections nav item.

```tsx
import { getConnectionHealthSummaries } from '@/lib/connections/health';
import type { ConnectionHealth } from '@/types/providers';

const HEALTH_COLORS: Record<ConnectionHealth, string> = {
  green: 'bg-emerald-500',
  amber: 'bg-amber-500',
  red: 'bg-red-500',
};

const PLATFORM_ORDER = ['facebook', 'instagram', 'gbp'] as const;

export async function ConnectionHealthDots(): Promise<JSX.Element> {
  let summaries;
  try {
    summaries = await getConnectionHealthSummaries();
  } catch {
    // If query fails (e.g., no account yet), show nothing
    return <></>;
  }
  
  if (!summaries.length) return <></>;
  
  const healthByPlatform = new Map(summaries.map(s => [s.provider, s.health]));
  
  return (
    <span className="inline-flex items-center gap-1 ml-2" aria-label="Connection health status">
      {PLATFORM_ORDER.map(platform => {
        const health = healthByPlatform.get(platform);
        if (!health) return null;
        return (
          <span
            key={platform}
            className={`inline-block h-2 w-2 rounded-full ${HEALTH_COLORS[health]}`}
            title={`${platform}: ${health}`}
            aria-label={`${platform} connection ${health}`}
          />
        );
      })}
    </span>
  );
}
```

**src/features/connections/connection-toast.tsx** -- Login toast for unhealthy connections (per D-03):

Client component that checks connection health on mount and shows a toast if any connection is amber or red. One-time per session using sessionStorage.

```tsx
'use client';

import { useEffect } from 'react';
import { toast } from 'sonner';
import type { ConnectionHealthSummary } from '@/types/providers';

interface ConnectionHealthToastProps {
  summaries: ConnectionHealthSummary[];
}

const PLATFORM_LABELS: Record<string, string> = {
  facebook: 'Facebook',
  instagram: 'Instagram',
  gbp: 'Google Business Profile',
};

const SESSION_KEY = 'cheersai_connection_toast_shown';

export function ConnectionHealthToast({ summaries }: ConnectionHealthToastProps): null {
  useEffect(() => {
    // One-time per session (per D-03)
    if (typeof window === 'undefined') return;
    if (sessionStorage.getItem(SESSION_KEY)) return;
    
    const unhealthy = summaries.filter(s => s.health === 'amber' || s.health === 'red');
    if (!unhealthy.length) return;
    
    sessionStorage.setItem(SESSION_KEY, 'true');
    
    // Show toast for each unhealthy connection (max 3)
    unhealthy.slice(0, 3).forEach(conn => {
      const label = PLATFORM_LABELS[conn.provider] ?? conn.provider;
      const message = conn.health === 'red'
        ? `${label} connection expired -- reconnect now`
        : `${label} token expires soon -- reconnect`;
      
      toast.warning(message, {
        action: {
          label: 'Reconnect',
          onClick: () => { window.location.href = '/connections'; },
        },
        duration: 8000,
      });
    });
  }, [summaries]);
  
  return null;
}
```
