    **src/lib/auth/rate-limit.ts:**
    Rebuild with Upstash per RESEARCH.md Pattern 3 and AUTH-08:
    ```typescript
    import { Ratelimit } from '@upstash/ratelimit';
    import { Redis } from '@upstash/redis';

    const redis = new Redis({
      url: process.env.UPSTASH_REDIS_REST_URL!,
      token: process.env.UPSTASH_REDIS_REST_TOKEN!,
    });

    export const authRateLimiter = new Ratelimit({
      redis,
      limiter: Ratelimit.slidingWindow(5, '60 s'),
      prefix: 'ratelimit:auth',
    });

    export async function checkAuthRateLimit(
      identifier: string
    ): Promise<{ allowed: boolean; limit: number; remaining: number; resetAt: number }> {
      const { success, limit, remaining, reset } = await authRateLimiter.limit(identifier);
      return { allowed: success, limit, remaining, resetAt: reset };
    }
    ```

    Add graceful fallback: if `UPSTASH_REDIS_REST_URL` is not set, `checkAuthRateLimit` returns `{ allowed: true, ... }` (skip rate limiting in dev). Log a warning when skipping.

    `export async function sendMagicLink(formData: FormData): Promise<{ success?: boolean; error?: string }>`:
    - Extract email from formData

    - Validate both with Zod
    - Check rate limit via `checkAuthRateLimit(email)`
    - Call `supabase.auth.signInWithPassword({ email, password })`
    - Return success/error
