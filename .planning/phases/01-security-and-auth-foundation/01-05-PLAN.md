    ```yaml
    name: CI

              path: node_modules

      typecheck:
        needs: install
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: '20'
          - uses: actions/cache/restore@v4
            with:
              path: node_modules
              key: modules-${{ hashFiles('package-lock.json') }}
          - run: npm run typecheck

      lint:
        needs: install
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: '20'
          - uses: actions/cache/restore@v4
            with:
              path: node_modules
              key: modules-${{ hashFiles('package-lock.json') }}
          - run: npm run lint -- --max-warnings=0

      test:
        needs: install
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: '20'
          - uses: actions/cache/restore@v4
            with:
              path: node_modules
              key: modules-${{ hashFiles('package-lock.json') }}
          - run: npx vitest run --coverage
        env:
          NODE_ENV: test

      build:
        needs: [typecheck, lint, test]
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: '20'
          - uses: actions/cache/restore@v4
            with:
              path: node_modules
              key: modules-${{ hashFiles('package-lock.json') }}
          - run: npm run build
            env:
              NEXT_PUBLIC_SUPABASE_URL: https://placeholder.supabase.co
              NEXT_PUBLIC_SUPABASE_ANON_KEY: placeholder-key
              # Other required public env vars for build
              NEXT_PUBLIC_SITE_URL: https://placeholder.cheersai.com

      migration-check:
        needs: install
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: '20'
          - uses: supabase/setup-cli@v1
          - run: supabase db lint --schema public
    ```

    - Coverage report generated in test job
    - migration-check uses `supabase db lint` per TEST-06
    - Placeholder env vars for build job (Next.js needs NEXT_PUBLIC_* at build time)
