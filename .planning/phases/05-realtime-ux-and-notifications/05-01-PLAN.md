Uses `payload.new.platform` for platform field. Generates plain-English message like "Post published to Facebook" or "Post failed on Instagram".
    - `mapNotificationToFeedEvent`: reads `payload.new.category`, `payload.new.message` (falling back to `payload.new.title`), `payload.new.metadata`, `payload.new.created_at` for timestamp, and `payload.new.read_at`.

    - Subscribes to `publish_jobs` UPDATE events on channel `failures:${accountId}`
    - If `payload.new.status === 'failed'`, increment count
    - If `payload.old?.status === 'failed' && payload.new.status !== 'failed'`, decrement count (min 0)
    - Returns current count

    ```typescript
    export async function getFailedPublishCount(): Promise<number> {
      const { supabase, accountId } = await requireAuthContext();
      const { count, error } = await supabase
        .from('publish_jobs')
        .select('*', { count: 'exact', head: true })
        .eq('account_id', accountId)
        .eq('status', 'failed');
      if (error) return 0;
      return count ?? 0;
    }

    export async function getUnreadNotificationCount(): Promise<number> {
      const { supabase, accountId } = await requireAuthContext();
      const { count, error } = await supabase
        .from('notifications')
        .select('*', { count: 'exact', head: true })
        .eq('account_id', accountId)
        .is('read_at', null)
        .is('dismissed_at', null);
      if (error) return 0;
      return count ?? 0;
    }
    ```
